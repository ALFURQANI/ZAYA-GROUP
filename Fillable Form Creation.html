from flask import Flask, render_template, request, send_file
import qrcode
from reportlab.pdfgen import canvas
import os

app = Flask(__name__)

@app.route('/')
def index():
    return render_template('form.html')

@app.route('/generate', methods=['POST'])
def generate():
    name = request.form['name']
    passport = request.form['passport']
    job = request.form['job']

    file_path = f"{name}.pdf"
    c = canvas.Canvas(file_path)

    c.drawString(100, 800, "ZAYA GROUP APPLICATION")
    c.drawString(100, 770, f"Name: {name}")
    c.drawString(100, 750, f"Passport: {passport}")
    c.drawString(100, 730, f"Job: {job}")

    # QR
    qr = qrcode.make(name + passport)
    qr.save("qr.png")
    c.drawImage("qr.png", 400, 700, width=100, height=100)

    c.save()

    return send_file(file_path, as_attachment=True)

if __name__ == '__main__':
    app.run(debug=True)

from flask import Flask, render_template, request, redirect, send_file
import sqlite3, qrcode, os
from reportlab.pdfgen import canvas

app = Flask(__name__)
UPLOAD_FOLDER = "uploads"

# DB
def connect():
    return sqlite3.connect("database.db")

# HOME
@app.route('/')
def dashboard():
    db = connect()
    data = db.execute("SELECT * FROM candidates").fetchall()
    return render_template("dashboard.html", data=data)

# FORM
@app.route('/form')
def form():
    return render_template("form.html")

# SAVE DATA
@app.route('/save', methods=['POST'])
def save():
    name = request.form['name']
    passport = request.form['passport']
    job = request.form['job']

    app_id = "ZG-DW-2025-0001"

    db = connect()
    db.execute("INSERT INTO candidates (name, passport, job, app_id) VALUES (?,?,?,?)",
               (name, passport, job, app_id))
    db.commit()

    return redirect('/')

# SAVE DATA
@app.route('/save', methods=['POST'])
def save():
    name = request.form['name']
    passport = request.form['passport']
    nationality = request.form['nationality']
    job = request.form['job']
    experience = request.form['experience']

    app_id = f"ZG-DW-2025-{str(len(name)+len(passport)).zfill(4)}"

    db = connect()
    db.execute("""INSERT INTO candidates 
        (name, passport, nationality, job, experience, app_id)
        VALUES (?,?,?,?,?,?)""",
        (name, passport, nationality, job, experience, app_id))
    db.commit()

    return redirect('/')

# SEARCH
@app.route('/search', methods=['POST'])
def search():
    keyword = request.form['keyword']

    db = connect()
    data = db.execute("""SELECT * FROM candidates 
        WHERE name LIKE ? OR passport LIKE ? OR app_id LIKE ?""",
        (f'%{keyword}%', f'%{keyword}%', f'%{keyword}%')).fetchall()

    return render_template("dashboard.html", data=data)

# DELETE
@app.route('/delete/<int:id>')
def delete(id):
    db = connect()
    db.execute("DELETE FROM candidates WHERE id=?", (id,))
    db.commit()
    return redirect('/')

# PDF GENERATION

    db = connect()
